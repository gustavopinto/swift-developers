AnswerCount,CommentCount,FavoriteCount,Tags

where (Tags like '%<swift>%' or

       Tags like '%<swift-playground>%' or
       Tags like '%<swift-json>%' or       
       Tags like '%<swift-extensions>%' or       
       Tags like '%<sqlite.swift>%' or

       Tags like '%<swift2.0>%' or
       Tags like '%<swift-custom-framework>%')
       AND CreationDate >= '2014-01-01'
       AND CreationDate <= '2014-12-31'
